            function makeAPIcall(url, requestBody){
                const body = {
                    url: url
                };


                return new Promise(function(resolve, reject){
                    $.post("http://localhost:8888/deezer", body, (data, status) => {
                })
                    .done(function(data){
                        resolve(data);
                    })
                    .fail(function(error) {
                        reject(error)
                    })
                    .always(function() {
                        console.log("API call has Finished.");
                    });
                });
            }
